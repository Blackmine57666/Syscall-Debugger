cmake_minimum_required(VERSION 3.17)
project(SyscallDebugger)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ../final)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -g")

file( GLOB_RECURSE SOURCES src/*.cpp )
file( GLOB_RECURSE HEADERS headers/*.h )
file( GLOB_RECURSE RES res/* )
add_executable(SyscallDebugger ${SOURCES} ${HEADERS} ${RES})

find_package(Qt5Widgets REQUIRED)
target_link_libraries(SyscallDebugger Qt5::Widgets)
target_link_libraries(SyscallDebugger pthread)
target_link_libraries(SyscallDebugger libexplain.so)
target_include_directories(SyscallDebugger PUBLIC headers)
